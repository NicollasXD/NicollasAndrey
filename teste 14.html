Conversor de Temperatura

Converte valores entre Celsius, Fahrenheit e Kelvin.

Ver Código
def celsius_para_fahrenheit(c): return (c * 9/5) + 32 def fahrenheit_para_celsius(f): return (f - 32) * 5/9 def celsius_para_kelvin(c): return c + 273.15 def kelvin_para_celsius(k): return k - 273.15 def main(): print("Conversor de Temperatura") temp = float(input("Digite a temperatura: ")) escala = input("Digite a escala original (C, F, K): ").upper() alvo = input("Para qual escala deseja converter? (C, F, K): ").upper() if escala == "C": if alvo == "F": print(f"{temp}°C = {celsius_para_fahrenheit(temp):.2f}°F") elif alvo == "K": print(f"{temp}°C = {celsius_para_kelvin(temp):.2f}K") else: print("Temperatura na mesma escala:", temp) elif escala == "F": if alvo == "C": print(f"{temp}°F = {fahrenheit_para_celsius(temp):.2f}°C") else: print("Conversão não implementada") elif escala == "K": if alvo == "C": print(f"{temp}K = {kelvin_para_celsius(temp):.2f}°C") else: print("Conversão não implementada") else: print("Escala inválida") if __name__ == "__main__": main()

Lista de Tarefas

Permite adicionar, listar e remover tarefas, com persistência em arquivo.

Ver Código
tarefas = [] def listar(): for i, tarefa in enumerate(tarefas): print(f"{i + 1}. {tarefa}") def adicionar(tarefa): tarefas.append(tarefa) print("Tarefa adicionada!") def remover(indice): try: tarefas.pop(indice - 1) print("Tarefa removida!") except: print("Índice inválido") def main(): while True: print("1. Listar\n2. Adicionar\n3. Remover\n4. Sair") opcao = input("Escolha: ") if opcao == '1': listar() elif opcao == '2': adicionar(input("Nova tarefa: ")) elif opcao == '3': remover(int(input("N° da tarefa: "))) elif opcao == '4': break else: print("Opção inválida") if __name__ == "__main__": main()

Gerador de Senhas

Gera senhas seguras com letras, números e símbolos.

Ver Código
import random import string def gerar_senha(tamanho=12): caracteres = string.ascii_letters + string.digits + string.punctuation senha = ''.join(random.choice(caracteres) for _ in range(tamanho)) return senha print("Senha gerada:", gerar_senha())

Conversor de Moedas

Converte valores usando cotações reais obtidas via API.

Ver Código
import requests def converter(valor, de, para): url = f"https://api.exchangerate-api.com/v4/latest/{de.upper()}" resposta = requests.get(url) dados = resposta.json() taxa = dados['rates'][para.upper()] return valor * taxa print("Exemplo: R$100 em USD:", converter(100, 'BRL', 'USD'))
